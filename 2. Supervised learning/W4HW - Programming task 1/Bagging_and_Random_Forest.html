def ot_estimatorov(z):
    out = []
    for k in z:
        RFC = RandomForestClassifier(n_estimators=k, 
                                     max_features=8, 
                                     max_depth=None)
        crossval_RFC = cross_val_score(RFC, X, y, cv=10)
        out += [crossval_RFC.mean()]
    return out
